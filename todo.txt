let cardAnimCtx

onMounted(() => {
  cardAnimCtx = gsap.context(() => {
    const cards = gsap.utils.toArray('.card')

    cards.forEach((card) => {
      gsap.set('.card__image', { yPercent: 15 })
      gsap.set('.card__image_deco', { yPercent: 10 })
      gsap.set('.card__content', { yPercent: 20 })
      gsap.set('.card__title', { yPercent: 25 })

      ScrollTrigger.create({
        animation: card,
        trigger: card,
        start: 'top 80%',
        end: 'bottom 20%',
        scrub: 1,
        markers: true
      })

      card
        .to('.card__image', { yPercent: 0 })
        .to('.card__image_deco', { yPercent: 0 }, '<')
        .to('.card__content', { yPercent: 0 }, '<')
        .to('.card__title', { yPercent: 0 }, '<')
    })
  })
})

onUnmounted(() => {
  cardAnimCtx.revert()
})
